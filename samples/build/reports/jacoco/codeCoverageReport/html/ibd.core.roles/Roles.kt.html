package ibd.core.roles

enum class Roles(private val value: String) {

    ASSOCIATE("Associate"), SR_ASSOCIATE("Sr Associate"), MANAGER("Manager");

    companion object {
        fun from(role: String) = values().firstOrNull { it.value == role }
    }
}
